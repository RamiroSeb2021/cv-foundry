\cvsubsection{Proyectos destacados}
\begin{cvprojectskills}
\cvprojectrow{Saber 11 (ICFES)}{\textbf{Análisis sociodemográfico del desempeño:} exploración, depuración y análisis de factores sociodemográficos e institucionales asociados al puntaje, con visualizaciones e interpretación orientada a hallazgos para toma de decisiones.}
\cvprojectrow{Regresión lineal}{\textbf{Modelos con variables climáticas:} construcción e interpretación de modelos de regresión lineal para explicar una variable respuesta a partir de covariables climáticas, incluyendo diagnóstico de supuestos, outliers e influencia.}
\cvprojectrow{GLM Gamma + splines}{\textbf{Modelamiento no lineal y ponderado:} ajuste de modelos lineales generalizados (Gamma) con splines y ponderación, comparación de alternativas y evaluación de ajuste/interpretabilidad para escenarios de negocio.}
\cvprojectrow{ICFES: AE + RF}{\textbf{Predicción del puntaje final:} pipeline con autoencoders (representaciones latentes) y Random Forest para estimar puntajes, con evaluación por métricas y análisis de variables latentes; desarrollo en el contexto de proyectos con SAS y Python.}
\cvprojectrow{RAG normativa aeronáutica}{\textbf{Sistema de consulta con LangChain/LangGraph:} ingestión de PDFs regulatorios, embeddings y búsqueda vectorial para recuperación contextual (RAG), con generación de respuestas y trazabilidad a las fuentes.}
\cvprojectrow{QA psicológico (LLMs)}{\textbf{Asistente QA y evaluación de calidad:} prototipo de sistema de preguntas y respuestas con LLMs, con foco en evaluación de alucinaciones, control de calidad de respuestas y consideraciones de seguridad.}
\cvprojectrow{Detección de veracidad}{\textbf{Modelos secuenciales con video:} extracción de rasgos faciales (OpenFace/LibreFace) y entrenamiento de un modelo BiLSTM en PyTorch para clasificar patrones temporales asociados a veracidad/engaño.}
\cvprojectrow{ARPRO}{\textbf{Analítica para consumo vs presupuesto:} apoyo a consultoría en analítica de datos para control y seguimiento de consumo y presupuestos, estructuración de información y modelamiento para pronóstico y toma de decisiones.}
\end{cvprojectskills}
